Template Strings

Similar to other JavaScript templating engines, JavaScript now has native templating using Template Strings.

  
    var listItems = [
      {
        "href" : "/",
        "label" : "Home"
      },
      {
        "href" : "/about",
        "label" : "About"
      },
      {
        "href" : "/work",
        "label" : "Work"
      }
    ];

    var menuTemplate = function(listItems){
      let menuTemplate = `<nav><ul>`;
      listItems.forEach(listItem => {
        menuTemplate += `<li><a href="${listItem.href}">${listItem.label}</a></li>`
      });
      menuTemplate += `</ul></nav>`;
      return menuTemplate;
    };

    var renderedTemplate = menuTemplate(listItems);
    // outputs <nav><ul><li><a href="/">Home</a></li><li><a href="/about">About</a></li><li><a href="/work">Work</a></li></ul></nav>
    console.log(renderedTemplate);
