    function GetName(){
      return this.firstName + " " + this.lastName;
    }

    function GetAge(){
      return this.age;
    }

    var person = {
      firstName: "Bill",
      lastName: "Sussman",
      age: 33
    }

    GetName.call(person); // outputs "Bill Sussman"
    GetAge.call(person); // outputs 33

    GetName.apply(person, []); // outputs "Bill Sussman"
    GetAge.apply(person, []); // outputs 33
